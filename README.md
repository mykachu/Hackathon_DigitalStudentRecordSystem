# To clone the project
.. code:: sh
    git clone https://github.com/mykachu/Hackathon_DigitalStudentRecordSystem.git

# To install requirements (must have Node.JS)

.. code:: sh

    # Install Yarn first 
    npm install -g yarn

    # Install requirements
    yarn install

    # Or
    yarn

# To run projects
.. code:: sh

    yarn start
